  SUPABASE SQL — TABELLEN ERSTELLEN (falls neu aufsetzen)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  create table players (name text primary key, flag text, pw text);
  create table tips (id serial primary key, player text, game_id text, tip_h text, tip_a text);
  create table results (game_id text primary key, score_h text, score_a text);
  create table wmtips (player text primary key, team text);
  create table chat (id serial primary key, player text, flag text, message text, created_at timestamptz default now());

  alter table players enable row level security;
  alter table tips enable row level security;
  alter table results enable row level security;
  alter table wmtips enable row level security;
  alter table chat enable row level security;

  create policy "all" on players for all using (true) with check (true);
  create policy "all" on tips for all using (true) with check (true);
  create policy "all" on results for all using (true) with check (true);
  create policy "all" on wmtips for all using (true) with check (true);
  create policy "all" on chat for all using (true) with check (true);
